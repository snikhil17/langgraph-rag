def _control_static(
    ends: Union[tuple[str, ...], dict[str, str]],
) -> Sequence[tuple[str, Any, Optional[str]]]:
    if isinstance(ends, dict):
        return [(CHANNEL_BRANCH_TO.format(k), None, label) for k, label in ends.items()]
    else:
