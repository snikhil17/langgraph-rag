"""Generate a string representation of the task path."""
def task_path_str(tup: Union[str, int, tuple]) -> str:

    return (
        f"~{', '.join(task_path_str(x) for x in tup)}"
        if isinstance(tup, (tuple, list))
        else f"{tup:010d}"
        if isinstance(tup, int)
        else str(tup)
    )
