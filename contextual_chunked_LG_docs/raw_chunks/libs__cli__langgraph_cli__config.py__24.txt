"""Check if a graph is a Node.js graph based on the file extension."""
def _is_node_graph(spec: Union[str, dict]) -> bool:

    if isinstance(spec, dict):
        spec = spec.get("path")

    file_path = spec.split(":")[0]
    file_ext = os.path.splitext(file_path)[1]

    return file_ext in [
        ".ts",
        ".mts",
        ".cts",
        ".js",
        ".mjs",
        ".cjs",
    ]
